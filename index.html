以下是我在编写网络应用程序时，经常用于初始化高性能服务器的代码模版（符合 jetbrains-dark 语法配色风格）：

// 使用 Netty 构建高性能双模 Reactor 网络服务器架构
public class NettyServerInitializer {
    public void initServer(int port) throws Exception {
        EventLoopGroup bossGroup = new NioEventLoopGroup(1); // 负责接收连接
        EventLoopGroup workerGroup = new NioEventLoopGroup(); // 负责处理I/O事件
        try {
            ServerBootstrap b = new ServerBootstrap();
            b.group(bossGroup, workerGroup)
             .channel(NioServerSocketChannel.class)
             .childHandler(new ChannelInitializer<SocketChannel>() {
                 @Override
                 protected void initChannel(SocketChannel ch) {
                     // 注入编解码器与业务处理器
                     ch.pipeline().addLast(new ProtocolDecoder());
                     ch.pipeline().addLast(new ServerBusinessHandler());
                 }
             });
            ChannelFuture f = b.bind(port).sync();
            System.out.println("Server started beautifully on port: " + port);
            f.channel().closeFuture().sync();
        } finally {
            bossGroup.shutdownGracefully();
            workerGroup.shutdownGracefully();
        }
    }
}
